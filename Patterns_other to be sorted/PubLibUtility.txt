// @description utilities for indicator and strategy development 
library('PubLibUtility')

//////////
// SIZE //
//////////

// @function size to string conversion
// @param literal string  
// @returns const string
export size(string size) =>
    size == 'Tiny' ? size.tiny : size == 'Small' ? size.small : size == 'Normal' ? size.normal : size == 'Large' ? size.large : size == 'Huge' ? size.huge : size == 'Auto' ? size.auto : na

///////////
// TABLE //
///////////

// @function table position to string conversion
// @param literal string  
// @returns const string
export tab_pos(string tab_pos) =>
    tab_pos == 'Top Right' ? position.top_right : tab_pos == 'Top Center' ? position.top_center : tab_pos == 'Top Left' ? position.top_left : tab_pos == 'Bottom Right' ? position.bottom_right : tab_pos == 'Bottom Center' ? position.bottom_center : tab_pos == 'Bottom Left' ? position.bottom_left : tab_pos == 'Middle Right' ? position.middle_right : tab_pos == 'Middle Center' ? position.middle_center : tab_pos == 'Middle Left' ? position.middle_left : na

////////////////////
// TIME TO STRING //
////////////////////

// @function time to string conversion
// @param series int
// @returns literal string
export time_string(int time_in) =>
    str.tostring(dayofmonth(time_in), '00') + '.' + str.tostring(month(time_in), '00') + '.' + str.tostring(year(time_in), '0000')
