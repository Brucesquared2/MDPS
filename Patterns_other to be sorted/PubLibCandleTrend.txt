/////////////////////////////
// CANDLE TREND CONDITIONS //
/////////////////////////////

// @function candle higher high condition
// @returns bool
export chh() =>
    high > high[1]

// @function candle higher low condition
// @returns bool
export chl() =>
    low > low[1]

// @function candle lower high condition
// @returns bool
export clh() =>
    high < high[1]

// @function candle lower low condition
// @returns bool
export cll() =>
    low < low[1]

// @function candle double top condition
// @returns bool
export cdt() =>
    high == high[1]

// @function candle double bottom condition
// @returns bool
export cdb() =>
    low == low[1]

// @function green candle condition
// @returns bool
export gc() =>
    close >= open

// @function green candle higher high condition
// @returns bool
export gchh() =>
    high > high[1] and gc()

// @function green candle higher low condition
// @returns bool
export gchl() =>
    low > low[1] and gc()

// @function green candle lower high condition
// @returns bool
export gclh() =>
    high < high[1] and gc()

// @function green candle lower low condition
// @returns bool
export gcll() =>
    low < low[1] and gc()

// @function green candle double top condition
// @returns bool
export gcdt() =>
    high == high[1] and gc()

// @function green candle double bottom condition
// @returns bool
export gcdb() =>
    low == low[1] and gc()

// @function red candle condition
// @returns bool
export rc() =>
    close < open

// @function red candle higher high condition
// @returns bool
export rchh() =>
    high > high[1] and rc()

// @function red candle higher low condition
// @returns bool
export rchl() =>
    low > low[1] and rc()

// @function red candle lower high condition
// @returns bool

// @function red candle lower low condition
// @returns bool

    low < low[1] and rc()

// @function red candle double top condition
// @returns bool
export rcdt() =>
    high == high[1] and rc()

// @function red candle double bottom condition
// @returns bool
export rcdb() =>
    low == low[1] and rc()
